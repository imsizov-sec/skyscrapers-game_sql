Тест 2: -- Требуется удаление сегодняшних побед (делать из-под создателя пакетов, т.е. администратора игры)

BEGIN
    DELETE FROM GAME_STEPS WHERE SESSION_ID IN (
        SELECT ID FROM GAME_SESSIONS 
        WHERE TRUNC(START_TIME) = TRUNC(SYSDATE) 
        AND STATUS_ID = (SELECT ID FROM GAME_STATUSES WHERE NAME = 'Победа')
    );
    DELETE FROM GAME_SESSIONS 
    WHERE TRUNC(START_TIME) = TRUNC(SYSDATE) 
    AND STATUS_ID = (SELECT ID FROM GAME_STATUSES WHERE NAME = 'Победа');
    COMMIT;
END;
